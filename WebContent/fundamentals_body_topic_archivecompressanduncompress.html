1. Tar is archival utility

    [root@susi susi]# mkdir dir1
    [root@susi susi]# cd dir1
    [root@susi dir1]# touch {fil1,fil2,fil3,fil4}
    [root@susi dir1]# ls
    fil1 fil2 fil3 fil4
    [root@susi dir1]#
    [root@susi dir1]# cd ../
    [root@susi susi]# pwd
    /home/susi
    [root@susi susi]# touch test1 test2
    [root@susi susi]#

  2. #gzip

    [root@susi susi]# ls
    dir1 test1 test2
    [root@susi susi]# gzip test1
    [root@susi susi]# ls
    dir1 test1.gz test2
    [root@susi susi]#

  3. Uncompress

    [root@susi susi]# ls
    dir1 test1.gz test2
    [root@susi susi]# gunzip test1.gz
    [root@susi susi]#

  4. Compress the directory with content

    [root@susi susi]# pwd
    /home/susi
    [root@susi susi]# ls
    dir1 test1 test2
    [root@susi susi]# tar -cvf dir.tar dir1/ test1 test2
    dir1/
    dir1/fil1
    dir1/fil2
    dir1/fil3
    dir1/fil4
    test1
    test2
    [root@susi susi]# ls
    dir1 dir.tar test1 test2
    [root@susi susi]#

    Check the content inside the archived file

    [root@susi susi]# tar -tf dir.tar
    dir1/
    dir1/fil1
    dir1/fil2
    dir1/fil3
    dir1/fil4
    test1
    test2
    [root@susi susi]#

    Untar the archival file

    [root@susi susi]# ls
    dir.tar
    [root@susi susi]# tar -xvf dir.tar
    dir1/
    dir1/fil1
    dir1/fil2
    dir1/fil3
    dir1/fil4
    test1
    test2
    [root@susi susi]# ls
    dir1 dir.tar test1 test2
    [root@susi susi]#

  5. Archive with gzip format

    [root@susi susi]# ls
    dir1 dir.tar test1 test2
    [root@susi susi]# tar -czf dir.tar.gz dir1/
    [root@susi susi]# ls
    dir1 dir.tar dir.tar.gz test1 test2
    [root@susi susi]#

    Unarchive gzip file

    [root@susi susi]# ls
    dir1 dir.tar.gz test1 test2
    [root@susi susi]# rm -rf dir1
    [root@susi susi]#
    [root@susi susi]# tar -xzf dir.tar.gz
    [root@susi susi]# ls
    dir1 dir.tar.gz test1 test2
    [root@susi susi]#

  6. Unzip and extract gunzip file

    [root@susi susi]# ls
    dir1 dir.tar.gz test1 test2
    [root@susi susi]# gzip test1
    [root@susi susi]# ls
    dir1 dir.tar.gz test1.gz test2
    [root@susi susi]# gunzip test1.gz
    [root@susi susi]# ls
    dir1 dir.tar.gz test1 test2
    [root@susi susi]#
